74  double rij = std::sqrt(xij[0]*xij[0]+xij[1]*xij[1]+xij[2]*xij[2]);

76  if (rij==0) return 1;
77 
78  // smooth coefficients
79  double mm2=pars->mm2;
80  double epi=pars->epi;
81 
82  // mass parameters
83  double mimj = pi.getMass()*pj.getMass(); // m_i*m_j
84  double mmij;
85  if (mm2>0 && epi>0) {
86  // Wij = mm2 if m_i*m_j < epi*m'^2; 0 otherwise;
87  if (mimj<epi*mm2) mmij = mm2;
88  else mmij = 0;
89  }
90  else {
91  mmij = mimj; // Wij = m_i*m_j
92  }
93 
94  Pij = - mimj / rij; // Potential energy
95  Wij = mmij / rij; // Transformation coefficient
96

99  double mor3 = pj.getMass() / rij3;
100  Aij[0] = mor3 * xij[0];
101  Aij[1] = mor3 * xij[1];
102  Aij[2] = mor3 * xij[2];
103 
104  // dW/dr
105  mor3 = mmij / rij3;
106  pWij[0] = mor3 * xij[0];
107  pWij[1] = mor3 * xij[1];
108  pWij[2] = mor3 * xij[2];
109 
110  return 0;
