382  void setIterSeq(const int itermax=20, const int sequence=2, const int intpmax=0) {
383  exp_sequence = sequence;
384 
385  // delete binomial array
386  if (bin_index!=NULL) {
387  if (step!=NULL) {
388  for (int i=0; i<step[exp_itermax]; i++)
389  if (bin_index[i]!=NULL) delete[] bin_index[i];
390  delete[] bin_index;
391  }
392  }
393 
394  // reset step array
395  if (step!=NULL) delete[] step;
396  step = new int[itermax+1];
397 
398  // calculate sequences of steps
399  // Romberg (even) sequence {h, h/2, h/4, h/8 ...}
400  if (sequence==1) EP::seq_Romberg(step,itermax+1);
401  // Bulirsch & Stoer sequence {h, h/2, h/3, h/4, h/6, h/8 ...}
402  else if (sequence==2) EP::seq_BS(step,itermax+1);
403  // E. Hairer (4k) sequences {h, h/2, h/6, h/10, h/14 ...}
404  else if (sequence==3) EP::seq_Hairer(step,itermax+1);
405  // Harmonic sequences {h, h/2, h/3, h/4 ...}
406  else EP::seq_Harmonic(step,itermax+1);
407 
408  // calculate binomial coefficients
409  bin_index = new int*[step[itermax]];
410  for (int i=0; i<step[itermax]; i++) {
411  bin_index[i] = new int[i+1];
412  if (i>0) EP::binomial_recursive_generator(bin_index[i],bin_index[i-1],i+1);
413  else EP::binomial_recursive_generator(bin_index[i],NULL,i+1);
414  }
415  exp_itermax = itermax;
416 
417  if (intpmax>itermax) {
418  std::cerr<<"Error: dense output interpolation derivate index ("<<intpmax<<") cannot be larger than extrapolation maximum sequence index ("<<itermax<<")!\n";
419  abort();
420  }
421  else if(intpmax==0) den_intpmax=itermax/2;
422  else den_intpmax = intpmax;
423 
424  }
