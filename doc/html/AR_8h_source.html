00857   /* Romberg_recursion_formula
00858      function: Using Romberg function: T_ik = T_i,k-1 + (T_i,k-1 - T_i-1,k-1) / [(h_i-k/h_i)^2 -1]
00859      argument: ti1k1: t_i-1,k-1
00860                tik1: t_i,k-1
00861                hr: h_i-k/h_i
00862      return:   T_ik
00863   */
00864   double Romberg_recursion_formula(const double ti1k1, const double tik1, const double hr) {
00865     if (hr==1) {
00866       std::cerr<<"Error!: h_i-k/h_i should not be 1.0! (Romberg_recursion_formula)";
00867       abort();
00868     }
00869     return tik1 + (tik1 - ti1k1)/(hr*hr - 1);
00870   }

00872   /* Rational_recursion formula
00873      function: Using rational function: T_ik = T_i,k-1 + (T_i,k-1 - T_i-1,k-1) / {(h_i-k/h_i)^2 * [1- (T_i,k-1 - T_i-1,k-1)/(T_i,k-1 - T_i-1,k-2)] -1}
00874      argument: ti1k2: T_i-1,k-2
00875                ti1k1: T_i-1,k-1
00876                tik1:  T_i,k-1
00877                hr:    h_i-k/h_i
00878      return:   T_ik
00879    */
00880   double Rational_recursion_formula(const double ti1k2, const double ti1k1, const double tik1, const double hr) {
00881     double dt2 = tik1 - ti1k2;
00882     double dt1 = tik1 - ti1k1;
00883     if (dt2==0) {
00884       if (dt1-dt2==0) return ti1k1;
00885       else return tik1;
00886     }
00887     else return tik1 + dt1/(hr*hr * (1 - dt1/dt2) - 1);
00888   }
